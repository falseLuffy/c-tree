### 基于vue开发的树形图

### 可选参数
```
   data:Array  [{name:'',children:[]}]
   property:Object  {name:'name',children:'children'}
   icon:
```
### 使用方法：
```
   <c-tree :data="data" :property="property"></tree>

   import CTree from './c-tree.js';

   {
    components:{
       CTree
    }
   }
```
   或
```
   <script src="./c-tree.js"></script>
   <script>

              data:[
                      {
                         name: '开始开始开',

                            {
                               name: '开始2',

                                  {
```
